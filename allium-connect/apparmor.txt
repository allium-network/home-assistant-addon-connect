# AppArmor profile for the Allium Connect (Home Node) add-on.
#
# Allium Connect is a privileged networking add-on: it brings up a WireGuard
# TUN interface, installs iptables NAT/forward chains (subnet router + Home
# exit), can run an embedded Tor client, and opens raw sockets. Tightly
# confining file / network / capabilities here would break those data paths,
# so this profile grants them broadly. Its purpose is the named-profile
# boundary itself (deny-by-default for anything not covered by the abstractions
# below, plus a place to tighten each path over time) — and it lifts the
# Supervisor security rating, which docks add-ons that ship no profile.
#
# The profile name MUST equal the add-on slug ("allium-connect").
#include <tunables/global>

profile allium-connect flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # docker-init (init: true) and the bundled tools signal each other.
  signal,

  # Broad file, network and capability access — see header. NET_ADMIN (routes,
  # iptables, TUN config), NET_RAW (raw sockets), and mknod (creating
  # /dev/net/tun when the Supervisor maps the dir but not the node) are the
  # load-bearing capabilities; the rest of the set is granted rather than
  # enumerated so a missing entry never silently breaks the mesh.
  file,
  network,
  capability,

  # WireGuard data path.
  /dev/net/tun rw,

  # Run the entrypoint, the add-on binary and the bundled CLI tools (bash, jq,
  # curl, ip, iptables, iptables-legacy, tor) in-profile.
  /run.sh rix,
  /usr/local/bin/allium-connect rix,
  /usr/local/sbin/** rix,
  /usr/bin/** rix,
  /usr/sbin/** rix,
  /bin/** rix,
  /sbin/** rix,

  # Persistent add-on data (token, WireGuard key, Ed25519 identity, tor dir).
  /data/** rwk,
}
